function min(a,b) { return a == 0 ? (b == 0 ? 101 : b) : (a <= b ? a : (b == 0 ? 101 : b)); }
function max(a,b) { return a >= b ? a : b; }
function min3(a,b,c) { return min(min(a,b),c); }
function max3(a,b,c) { return max(max(a,b),c); }
